Table Parking {
  parking_Spot_id integer [primary key]
  Parking_spot_no varchar
  floor_no integer
  spot_type varchar
  is_available boolean
}

Table vehicle {
  vehicle_id integer [primary key]
  vehicle_num varchar2
  vehicle_type varchar
}

Table ticket {
  ticket_id integer unique [primary key]
  entry_time timestamp
  exit_time timestamp
}

Table fees{
  ticket_id integer
  vehicle_id integer
  parking_Spot_id integer
  entry_time timestamp
  exit_time timestamp
  total_price integer

}
